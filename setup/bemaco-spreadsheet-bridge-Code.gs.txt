function doGet(e) {
  const callback = String((e && e.parameter && e.parameter.callback) || 'personnelBridgeCallback');

  try {
    validateBridgeToken_((e && e.parameter && e.parameter.token) || '');
    const force = String((e && e.parameter && e.parameter.force) || '') === '1';
    const cachedSheets = force ? null : getCachedSheets_();
    if (cachedSheets) {
      return javascriptResponse_(callback, {
        ok: true,
        sheets: cachedSheets,
        cache: 'hit',
        updatedAt: new Date().toISOString()
      });
    }

    const spreadsheet = getBoundSpreadsheet_();
    const sheets = {};
    Object.keys(PERSONNEL_SHEETS).forEach(function(sourceId) {
      const sheet = getOrCreateSheet_(spreadsheet, sourceId);
      ensureSourceHeaders_(sheet, sourceId, {});
      sheets[sourceId] = readSheet_(sheet);
    });
    setCachedSheets_(sheets);

    return javascriptResponse_(callback, {
      ok: true,
      sheets: sheets,
      cache: 'miss',
      updatedAt: new Date().toISOString()
    });
  } catch (error) {
    return javascriptResponse_(callback, {
      ok: false,
      error: error.message || String(error)
    });
  }
}

function doPost(e) {
  try {
    const payload = JSON.parse((e && e.postData && e.postData.contents) || '{}');
    validateBridgeToken_(payload.token);

    const firebaseUser = verifyFirebaseUser_(payload.firebaseIdToken);
    const role = getFirebaseRole_(firebaseUser.email, payload.firebaseIdToken);
    if (PERSONNEL_WRITE_ROLES.indexOf(role) === -1) {
      throw new Error('Role ' + role + ' tidak diizinkan mengubah data spreadsheet.');
    }

    const sourceId = String(payload.sourceId || '');
    const targetSourceId = String(payload.targetSourceId || sourceId);
    assertSource_(sourceId);
    assertSource_(targetSourceId);

    const spreadsheet = getBoundSpreadsheet_();
    const sourceSheet = getOrCreateSheet_(spreadsheet, sourceId);
    const targetSheet = getOrCreateSheet_(spreadsheet, targetSourceId);
    const action = String(payload.action || '');
    const data = addAuditFields_(payload.data || {}, firebaseUser.email);

    ensureSourceHeaders_(sourceSheet, sourceId, data);
    ensureSourceHeaders_(targetSheet, targetSourceId, data);

    const lock = LockService.getDocumentLock();
    lock.waitLock(30000);

    try {
      if (action === 'ensureSheet') {
        ensureSourceHeaders_(targetSheet, targetSourceId, data);
      } else if (action === 'add') {
        addRow_(targetSheet, data);
      } else if (action === 'update') {
        if (targetSourceId !== sourceId) {
          addRow_(targetSheet, data);
          deleteRow_(sourceSheet, Number(payload.rowNumber));
        } else {
          updateRow_(sourceSheet, Number(payload.rowNumber), data);
        }
      } else if (action === 'upsert') {
        upsertRow_(targetSheet, data, payload.matchColumn || '', payload.matchValue || '');
      } else if (action === 'deleteByKey') {
        deleteRowByKey_(sourceSheet, payload.matchColumn || '', payload.matchValue || '');
      } else if (action === 'delete') {
        deleteRow_(sourceSheet, Number(payload.rowNumber));
      } else {
        throw new Error('Aksi spreadsheet tidak dikenali.');
      }
      clearBridgeCache_();
    } finally {
      lock.releaseLock();
    }

    return jsonResponse_({
      ok: true,
      action: action,
      sourceId: sourceId,
      targetSourceId: targetSourceId,
      email: firebaseUser.email,
      role: role
    });
  } catch (error) {
    console.error(error);
    return jsonResponse_({
      ok: false,
      error: error.message || String(error)
    });
  }
}

function getOrCreateSheet_(spreadsheet, sourceId) {
  const config = PERSONNEL_SHEETS[sourceId];
  if (!config) throw new Error('Sumber sheet tidak valid: ' + sourceId);

  const gid = Number(config.gid || 0);
  let sheet = spreadsheet.getSheets().find(function(item) {
    return gid && item.getSheetId() === gid;
  });

  if (!sheet && Array.isArray(config.names)) {
    sheet = spreadsheet.getSheets().find(function(item) {
      const currentName = normalizeName_(item.getName());
      return config.names.some(function(candidate) {
        return currentName === normalizeName_(candidate);
      });
    });
  }

  if (!sheet) {
    sheet = spreadsheet.insertSheet(config.names[0]);
    ensureSourceHeaders_(sheet, sourceId, {});
  }

  return sheet;
}

function ensureSourceHeaders_(sheet, sourceId, data) {
  const defaultHeaders = PERSONNEL_SHEETS[sourceId].headers || [];
  const dataHeaders = Object.keys(data || {}).filter(function(key) {
    return key && key.charAt(0) !== '_' && !isComputedHeader_(key);
  });
  const desiredHeaders = mergeHeaders_(defaultHeaders, dataHeaders);
  if (!desiredHeaders.length) return;

  const values = getSheetValues_(sheet);
  const hasContent = values.some(function(row) {
    return row.some(function(value) { return String(value || '').trim() !== ''; });
  });
  const headerIndex = hasContent ? findHeaderRow_(values) : 0;
  let headers = hasContent ? makeUniqueHeaders_(values[headerIndex]) : [];
  const genericHeaders = headers.length && headers.every(function(header) {
    return /^column\s+\d+$/i.test(String(header || '').trim());
  });

  if (!headers.length || genericHeaders || !hasContent) {
    writeHeaderRow_(sheet, desiredHeaders, 1);
    return;
  }

  const additions = desiredHeaders.filter(function(header) {
    return headers.indexOf(header) === -1;
  });
  if (!additions.length) return;

  const startColumn = headers.length + 1;
  if (sheet.getMaxColumns() < headers.length + additions.length) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), headers.length + additions.length - sheet.getMaxColumns());
  }
  sheet.getRange(headerIndex + 1, startColumn, 1, additions.length).setValues([additions]);
}

function writeHeaderRow_(sheet, headers, rowNumber) {
  if (sheet.getMaxColumns() < headers.length) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), headers.length - sheet.getMaxColumns());
  }
  sheet.getRange(rowNumber, 1, 1, headers.length).setValues([headers]);
  sheet.setFrozenRows(rowNumber);
}

function readSheet_(sheet) {
  const values = getSheetValues_(sheet);
  if (!values.length) return [];

  const headerIndex = findHeaderRow_(values);
  const headers = makeUniqueHeaders_(values[headerIndex]);
  const records = [];

  for (let rowIndex = headerIndex + 1; rowIndex < values.length; rowIndex += 1) {
    const row = values[rowIndex];
    if (!row.some(function(value) { return String(value || '').trim() !== ''; })) continue;

    const record = { '_Sumber Baris': rowIndex + 1 };
    headers.forEach(function(header, columnIndex) {
      record[header] = row[columnIndex] || '';
    });
    records.push(record);
  }
  return records;
}

function addRow_(sheet, data) {
  const values = getSheetValues_(sheet);
  const headerIndex = findHeaderRow_(values);
  const headers = makeUniqueHeaders_(values[headerIndex]);
  const targetRow = Math.max(sheet.getLastRow() + 1, headerIndex + 2);
  const sourceFormulaRow = Math.max(headerIndex + 2, targetRow - 1);

  headers.forEach(function(header, columnIndex) {
    const column = columnIndex + 1;
    const cell = sheet.getRange(targetRow, column);
    const sourceFormula = sourceFormulaRow < targetRow ? sheet.getRange(sourceFormulaRow, column).getFormulaR1C1() : '';

    if (Object.prototype.hasOwnProperty.call(data, header) && !isComputedHeader_(header)) {
      cell.setValue(normalizeWriteValue_(data[header]));
    } else if (sourceFormula) {
      cell.setFormulaR1C1(sourceFormula);
    }
  });
}

function updateRow_(sheet, rowNumber, data) {
  const values = getSheetValues_(sheet);
  const headerIndex = findHeaderRow_(values);
  const headers = makeUniqueHeaders_(values[headerIndex]);
  const firstDataRow = headerIndex + 2;

  if (!Number.isInteger(rowNumber) || rowNumber < firstDataRow || rowNumber > sheet.getLastRow()) {
    throw new Error('Nomor baris spreadsheet tidak valid.');
  }

  headers.forEach(function(header, columnIndex) {
    if (!Object.prototype.hasOwnProperty.call(data, header) || isComputedHeader_(header)) return;
    const cell = sheet.getRange(rowNumber, columnIndex + 1);
    if (cell.getFormula()) return;
    cell.setValue(normalizeWriteValue_(data[header]));
  });
}

function upsertRow_(sheet, data, matchColumn, matchValue) {
  const values = getSheetValues_(sheet);
  const headerIndex = findHeaderRow_(values);
  const headers = makeUniqueHeaders_(values[headerIndex]);
  const matchHeader = String(matchColumn || '').trim() || findBestMatchColumn_(headers, data);
  const matchIndex = headers.indexOf(matchHeader);
  const expected = String(matchValue || data[matchHeader] || '').trim();

  if (matchIndex !== -1 && expected) {
    for (let rowIndex = headerIndex + 1; rowIndex < values.length; rowIndex += 1) {
      if (String(values[rowIndex][matchIndex] || '').trim() === expected) {
        updateRow_(sheet, rowIndex + 1, data);
        return;
      }
    }
  }

  addRow_(sheet, data);
}

function deleteRow_(sheet, rowNumber) {
  const values = getSheetValues_(sheet);
  const firstDataRow = findHeaderRow_(values) + 2;
  if (!Number.isInteger(rowNumber) || rowNumber < firstDataRow || rowNumber > sheet.getLastRow()) {
    throw new Error('Nomor baris spreadsheet tidak valid.');
  }
  sheet.deleteRow(rowNumber);
}

function deleteRowByKey_(sheet, matchColumn, matchValue) {
  const values = getSheetValues_(sheet);
  const headerIndex = findHeaderRow_(values);
  const headers = makeUniqueHeaders_(values[headerIndex]);
  const matchIndex = headers.indexOf(String(matchColumn || '').trim());
  const expected = String(matchValue || '').trim();
  if (matchIndex === -1 || !expected) return;

  for (let rowIndex = headerIndex + 1; rowIndex < values.length; rowIndex += 1) {
    if (String(values[rowIndex][matchIndex] || '').trim() === expected) {
      sheet.deleteRow(rowIndex + 1);
      return;
    }
  }
}

function getSheetValues_(sheet) {
  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastColumn = Math.max(sheet.getLastColumn(), 1);
  return sheet.getRange(1, 1, lastRow, lastColumn).getDisplayValues();
}

function findHeaderRow_(rows) {
  let bestIndex = 0;
  let bestScore = -1;
  const limit = Math.min(rows.length, 35);

  for (let index = 0; index < limit; index += 1) {
    const row = rows[index].map(function(value) {
      return String(value || '').trim();
    });
    const nonEmpty = row.filter(Boolean).length;
    const text = normalizeText_(row.join(' '));
    const keywordScore = [
      'id', 'nama', 'personil', 'jabatan', 'posisi', 'status', 'pekerjaan',
      'tanggal', 'termin', 'inventaris', 'finance', 'nilai'
    ].reduce(function(score, keyword) {
      return score + (text.indexOf(keyword) !== -1 ? 4 : 0);
    }, 0);
    const score = nonEmpty + keywordScore;

    if (score > bestScore) {
      bestScore = score;
      bestIndex = index;
    }
  }
  return bestIndex;
}

function makeUniqueHeaders_(row) {
  const counts = {};
  return row.map(function(value, index) {
    const base = String(value || '').trim() || 'Kolom ' + (index + 1);
    counts[base] = (counts[base] || 0) + 1;
    return counts[base] === 1 ? base : base + ' ' + counts[base];
  });
}

function mergeHeaders_(baseHeaders, extraHeaders) {
  const seen = {};
  return [].concat(baseHeaders || [], extraHeaders || []).filter(function(header) {
    const key = String(header || '').trim();
    if (!key || seen[key]) return false;
    seen[key] = true;
    return true;
  });
}

function findBestMatchColumn_(headers, data) {
  return ['Row Key', 'ID', 'ID Tender', 'ID Pekerjaan', 'Pekerjaan', 'Nama Pekerjaan']
    .find(function(header) {
      return headers.indexOf(header) !== -1 && data[header];
    }) || '';
}

function addAuditFields_(data, email) {
  const result = {};
  Object.keys(data || {}).forEach(function(key) {
    result[key] = data[key];
  });
  if (!result['Updated At']) result['Updated At'] = new Date().toISOString();
  if (!result['Updated By']) result['Updated By'] = email || '';
  if (!result['Row Key']) result['Row Key'] = Utilities.getUuid();
  return result;
}

function isComputedHeader_(header) {
  const normalized = normalizeText_(header);
  return [
    'pekerjaan aktif', 'tugas aktif', 'project aktif', 'keterlibatan pekerjaan',
    'status selesai', 'akumulasi'
  ].some(function(keyword) {
    return normalized.indexOf(keyword) !== -1;
  });
}

function normalizeWriteValue_(value) {
  return String(value == null ? '' : value).trim();
}

function normalizeText_(value) {
  return String(value || '').toLowerCase().replace(/[^a-z0-9]+/g, ' ').trim();
}

function normalizeName_(value) {
  return String(value || '').trim().toLowerCase();
}

function assertSource_(sourceId) {
  if (!Object.prototype.hasOwnProperty.call(PERSONNEL_SHEETS, sourceId)) {
    throw new Error('Sumber sheet tidak valid: ' + sourceId);
  }
}

function getCachedSheets_() {
  const sheets = {};
  const sourceIds = Object.keys(PERSONNEL_SHEETS);
  for (let index = 0; index < sourceIds.length; index += 1) {
    const sourceId = sourceIds[index];
    const cachedRecords = getCacheJson_('sheet:' + sourceId);
    if (!Array.isArray(cachedRecords)) return null;
    sheets[sourceId] = cachedRecords;
  }
  return sheets;
}

function setCachedSheets_(sheets) {
  Object.keys(PERSONNEL_SHEETS).forEach(function(sourceId) {
    if (Array.isArray(sheets[sourceId])) {
      setCacheJson_('sheet:' + sourceId, sheets[sourceId], PERSONNEL_CACHE_TTL_SECONDS);
    }
  });
}

function clearBridgeCache_() {
  Object.keys(PERSONNEL_SHEETS).forEach(function(sourceId) {
    clearCacheValue_('sheet:' + sourceId);
  });
}

function getCacheJson_(key) {
  const text = getCacheText_(key);
  if (!text) return null;
  try {
    return JSON.parse(text);
  } catch (error) {
    clearCacheValue_(key);
    return null;
  }
}

function setCacheJson_(key, value, ttlSeconds) {
  setCacheText_(key, JSON.stringify(value), ttlSeconds);
}

function getCacheText_(key) {
  const cache = CacheService.getScriptCache();
  const baseKey = PERSONNEL_CACHE_PREFIX + key;
  const metaText = cache.get(baseKey + ':meta');
  if (!metaText) return '';

  let meta;
  try {
    meta = JSON.parse(metaText);
  } catch (error) {
    clearCacheValue_(key);
    return '';
  }

  if (!meta || !Number.isInteger(meta.chunks) || meta.chunks < 1 || meta.chunks > PERSONNEL_CACHE_MAX_CHUNKS) {
    clearCacheValue_(key);
    return '';
  }

  const keys = [];
  for (let index = 0; index < meta.chunks; index += 1) {
    keys.push(baseKey + ':chunk:' + index);
  }

  const chunks = cache.getAll(keys);
  const parts = [];
  for (let index = 0; index < keys.length; index += 1) {
    const chunk = chunks[keys[index]];
    if (typeof chunk !== 'string') {
      clearCacheValue_(key);
      return '';
    }
    parts.push(chunk);
  }
  return parts.join('');
}

function setCacheText_(key, text, ttlSeconds) {
  const cache = CacheService.getScriptCache();
  const baseKey = PERSONNEL_CACHE_PREFIX + key;
  const chunks = [];

  for (let start = 0; start < text.length; start += PERSONNEL_CACHE_CHUNK_SIZE) {
    chunks.push(text.slice(start, start + PERSONNEL_CACHE_CHUNK_SIZE));
  }

  if (!chunks.length || chunks.length > PERSONNEL_CACHE_MAX_CHUNKS) {
    clearCacheValue_(key);
    return;
  }

  const payload = {};
  chunks.forEach(function(chunk, index) {
    payload[baseKey + ':chunk:' + index] = chunk;
  });
  payload[baseKey + ':meta'] = JSON.stringify({
    chunks: chunks.length,
    storedAt: new Date().toISOString()
  });
  cache.putAll(payload, ttlSeconds);
}

function clearCacheValue_(key) {
  const cache = CacheService.getScriptCache();
  const baseKey = PERSONNEL_CACHE_PREFIX + key;
  const keys = [baseKey + ':meta'];
  for (let index = 0; index < PERSONNEL_CACHE_MAX_CHUNKS; index += 1) {
    keys.push(baseKey + ':chunk:' + index);
  }
  cache.removeAll(keys);
}

function getBoundSpreadsheet_() {
  try {
    return SpreadsheetApp.openById(SPREADSHEET_ID);
  } catch (error) {
    throw new Error('Spreadsheet Bemaco tidak dapat dibuka: ' + error.message);
  }
}

function validateBridgeToken_(token) {
  if (!token || String(token) !== PERSONNEL_BRIDGE_TOKEN) {
    throw new Error('Token Apps Script Bridge tidak valid.');
  }
}

function verifyFirebaseUser_(idToken) {
  if (!idToken) throw new Error('Sesi Firebase tidak tersedia.');
  const response = UrlFetchApp.fetch(
    'https://identitytoolkit.googleapis.com/v1/accounts:lookup?key=' + encodeURIComponent(FIREBASE_API_KEY),
    {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify({ idToken: idToken }),
      muteHttpExceptions: true
    }
  );
  const result = JSON.parse(response.getContentText() || '{}');
  const user = result.users && result.users[0];
  if (response.getResponseCode() !== 200 || !user || !user.email) {
    throw new Error('Sesi Firebase tidak valid atau sudah berakhir.');
  }
  return {
    uid: user.localId,
    email: String(user.email).toLowerCase()
  };
}

function getFirebaseRole_(email, idToken) {
  if (email === BOOTSTRAP_SUPER_ADMIN_EMAIL) return 'super_admin';

  const roleUrl =
    'https://firestore.googleapis.com/v1/projects/' + FIREBASE_PROJECT_ID +
    '/databases/(default)/documents/roles/' + encodeURIComponent(email);
  const response = UrlFetchApp.fetch(roleUrl, {
    method: 'get',
    headers: { Authorization: 'Bearer ' + idToken },
    muteHttpExceptions: true
  });

  if (response.getResponseCode() === 404) return 'member';
  const result = JSON.parse(response.getContentText() || '{}');
  return result.fields && result.fields.role && result.fields.role.stringValue
    ? result.fields.role.stringValue
    : 'member';
}

function javascriptResponse_(callback, data) {
  const safeCallback = /^[A-Za-z_$][A-Za-z0-9_$.]*$/.test(callback)
    ? callback
    : 'personnelBridgeCallback';
  return ContentService
    .createTextOutput(safeCallback + '(' + JSON.stringify(data) + ');')
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}

function jsonResponse_(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
